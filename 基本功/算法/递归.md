# 递归

## 二叉树的递归遍历

```js
class createNode {
  constructor(val) {
    this.val = val;
    this.left = this.right = null;
  }
}
const nodea = new createNode("a");
const nodeb = new createNode("b");
const nodec = new createNode("c");
const noded = new createNode("d");
const nodee = new createNode("e");
const nodef = new createNode("f");
nodea.left = nodeb;
nodea.right = nodec;
nodeb.left = noded;
nodeb.right = nodee;
nodec.left = nodef;
        a
   b         c
d    e     f
```

### 先序遍历

```js
function preorder(node) {
  if (!node) {
    return;
  }
  console.log(node.val);
  preorder(node.left);
  preorder(node.right);
}
```

### 中序遍历

```js
function preorder(node) {
  if (!node) {
    return;
  }
  preorder(node.left);
  console.log(node.val);
  preorder(node.right);
}
```

### 后序遍历

```js
function preorder(node) {
  if (!node) {
    return;
  }
  preorder(node.left);
  preorder(node.right);
  console.log(node.val);
}
preorder(nodea);
```
